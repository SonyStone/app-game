declare module 'solid-js' {
  namespace JSX {
    interface Directives {
      autoFocus: boolean;
    }
  }
}

function autoFocus(el: HTMLElement, accessor: Accessor<boolean>) {
  createEffect(() => {
    if (accessor()) el.focus();
  });
}

// Usage
<input use:autoFocus={true} placeholder="Auto-focused" />
